Ember FormBuilder

A formbuilder for Ember.js

Modal panes

{{#formFor "object"}}
  {{#fieldsFor "books"}}
    {{input "title"}}
    {{removeAssociation "books" text="Remove Book" classes="btn btn-danger"}}
  {{/fieldsFor}}
  {{addAssociation "books" objectClass="Ember.Object" text="Add Book" classes="btn btn-success"}}
{{/formFor}}
